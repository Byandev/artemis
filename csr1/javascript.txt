const leaderboard = document.getElementById("leaderboard");

async function fetchData(period) {
    const res = await fetch(`/leaderboard/${period}`);
    return await res.json();
}

function formatPeso(value){
    return new Intl.NumberFormat('en-PH',{
        style:'currency',
        currency:'PHP'
    }).format(value)
}

function loadLeaderboard(dataset){
    leaderboard.innerHTML=""

    dataset.slice(3).forEach((user,index)=>{
        const row=document.createElement("div")
        row.classList.add("row")

        row.innerHTML=`
        <span>${index+4}</span>
        <span class="name">
        <img src="${user.avatar}">
        ${user.name}
        </span>
        <span>${user.orders}</span>
        <span>${formatPeso(user.sales)}</span>
        `

        leaderboard.appendChild(row)
    })
}

function updatePodium(dataset){
    const cards=document.querySelectorAll(".podium .card")

    cards[1].querySelector(".avatar").src=dataset[0].avatar
    cards[1].querySelector(".csr-name").textContent=dataset[0].name
    cards[1].querySelector(".orders").textContent="Orders: "+dataset[0].orders
    cards[1].querySelector(".sales").textContent=formatPeso(dataset[0].sales)

    cards[0].querySelector(".avatar").src=dataset[1].avatar
    cards[0].querySelector(".csr-name").textContent=dataset[1].name
    cards[0].querySelector(".orders").textContent="Orders: "+dataset[1].orders
    cards[0].querySelector(".sales").textContent=formatPeso(dataset[1].sales)

    cards[2].querySelector(".avatar").src=dataset[2].avatar
    cards[2].querySelector(".csr-name").textContent=dataset[2].name
    cards[2].querySelector(".orders").textContent="Orders: "+dataset[2].orders
    cards[2].querySelector(".sales").textContent=formatPeso(dataset[2].sales)
}

async function load(period){
    const data = await fetchData(period)
    loadLeaderboard(data)
    updatePodium(data)
}

const tabs=document.querySelectorAll(".tab")

tabs.forEach(tab=>{
    tab.addEventListener("click",()=>{
        tabs.forEach(t=>t.classList.remove("active"))
        tab.classList.add("active")

        const type=tab.textContent.toLowerCase()
        load(type)
    })
})

window.onload = () => {
    setTimeout(()=>{
        document.getElementById("loader").style.display="none";
        document.getElementById("main-content").style.display="block";
        load("daily");
    },2000)
}
