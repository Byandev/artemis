function getBaseOptions(isDark) {
  return {
    chart: {
      fontFamily: '"DM Sans", system-ui, sans-serif',
      toolbar: { show: false },
      zoom: { enabled: false },
      background: 'transparent',
      animations: {
        enabled: true, speed: 600,
        easing: 'easeinout',
        dynamicAnimation: { speed: 400 },
      },
    },
    colors: isDark ? CHART_COLORS_DARK : CHART_COLORS,
    grid: {
      borderColor: isDark
        ? 'rgba(255,255,255,0.04)' : 'rgba(0,0,0,0.04)',
      strokeDashArray: 0,
      xaxis: { lines: { show: false } },
      yaxis: { lines: { show: true } },
      padding: { top: 0, right: 4, bottom: 0, left: 8 },
    },
    xaxis: {
      axisBorder: { show: false },
      axisTicks: { show: false },
      labels: {
        style: {
          fontFamily: '"DM Mono", monospace',
          fontSize: '11px', fontWeight: 400,
          colors: isDark ? '#5C5C61' : '#9D9DA1',
        },
      },
    },
    yaxis: {
      labels: {
        style: {
          fontFamily: '"DM Mono", monospace',
          fontSize: '11px', fontWeight: 400,
          colors: isDark ? '#5C5C61' : '#9D9DA1',
        },
        formatter: (val) => formatCompact(val),
      },
    },
    tooltip: {
      theme: isDark ? 'dark' : 'light',
      style: { fontFamily: '"DM Sans", sans-serif', fontSize: '12px' },
      y: { formatter: (val) => formatCurrency(val) },
    },
    dataLabels: { enabled: false },
    legend: {
      fontFamily: '"DM Sans", sans-serif',
      fontSize: '12px', fontWeight: 500,
      labels: { colors: isDark ? '#8E8E93' : '#6B6B6F' },
      markers: { size: 4, shape: 'circle', offsetX: -2 },
      itemMargin: { horizontal: 16, vertical: 4 },
    },
    states: {
      hover: { filter: { type: 'darken', value: 0.05 } },
      active: { filter: { type: 'darken', value: 0.1 } },
    },
  };
}

function getLineAreaOptions(isDark) {
  const base = getBaseOptions(isDark);
  return {
    ...base,
    chart: { ...base.chart, type: 'area' },
    stroke: {
      curve: 'smooth',
      width: [2.5, 2],       // primary thicker
      dashArray: [0, 4],     // secondary dashed
    },
    fill: {
      type: 'gradient',
      gradient: {
        shadeIntensity: 1,
        opacityFrom: 0.2, opacityTo: 0,
        stops: [0, 100],
      },
    },
    markers: {
      size: 0,               // hidden by default
      hover: { size: 5, sizeOffset: 2 },
    },
    tooltip: {
      ...base.tooltip,
      shared: true, intersect: false,
      x: { format: 'MMM dd, yyyy' },
    },
    xaxis: { ...base.xaxis, type: 'datetime' },
  };
}

function getDonutOptions(isDark, totalLabel = 'Total') {
  return {
    chart: {
      type: 'donut',
      fontFamily: '"DM Sans", system-ui, sans-serif',
      background: 'transparent',
      animations: { enabled: true, speed: 600,
        animateGradually: { enabled: true, delay: 80 } },
    },
    colors: isDark ? CHART_COLORS_DARK : CHART_COLORS,
    stroke: {
      show: true, width: 2,
      colors: [isDark ? '#18181B' : '#FFFFFF'],
    },
    plotOptions: {
      pie: {
        donut: {
          size: '72%',
          labels: {
            show: true,
            name: {
              show: true, fontSize: '12px',
              fontFamily: '"DM Sans", sans-serif',
              fontWeight: 500,
              color: isDark ? '#8E8E93' : '#6B6B6F',
              offsetY: -8,
            },
            value: {
              show: true, fontSize: '22px',
              fontFamily: '"DM Mono", monospace',
              fontWeight: 600,
              color: isDark ? '#F0F0EC' : '#1A1A1D',
              offsetY: 4,
              formatter: (v) => Number(v).toLocaleString(),
            },
            total: {
              show: true, showAlways: true,
              label: totalLabel,
              fontSize: '11px',
              fontFamily: '"DM Sans", sans-serif',
              color: isDark ? '#5C5C61' : '#9D9DA1',
              formatter: (w) =>
                w.globals.seriesTotals
                  .reduce((a, b) => a + b, 0)

        },
      },
    },
    legend: {
      position: 'right',
      fontFamily: '"DM Sans"', fontSize: '12px',

// Use in: tooltip: { custom: buildCustomTooltip }

function buildCustomTooltip({ series, dataPointIndex, w }) {
  const isDark = document.documentElement.classList.contains('dark');
  const bg     = isDark ? '#222225' : '#FFFFFF';
  const border = isDark ? 'rgba(255,255,255,0.06)' : 'rgba(0,0,0,0.06)';
  const t1     = isDark ? '#F0F0EC' : '#1A1A1D';
  const t2     = isDark ? '#8E8E93' : '#6B6B6F';
  const t4     = isDark ? '#3A3A3E' : '#C5C5C9';
  const shadow  = isDark
    ? '0 8px 24px rgba(0,0,0,0.4)'
    : '0 8px 24px rgba(0,0,0,0.08)';

  const label = w.globals.categoryLabels?.[dataPointIndex]
    || w.globals.labels?.[dataPointIndex] || '';

  let rows = '';
  w.globals.seriesNames.forEach((name, i) => {
    const color = w.config.colors[i];
    const val = series[i]?.[dataPointIndex];
    if (val == null) return;
    rows += '<div style="display:flex;align-items:center;'
      + 'gap:8px;margin-top:4px">'
      + '<span style="width:6px;height:6px;border-radius:3px;'
      + 'background:' + color + '"></span>'
      + '<span style="font-size:12px;color:' + t2
      + ';flex:1">' + name + '</span>'
      + '<span style="font-size:12px;font-family:DM Mono,monospace;'
      + 'font-weight:500;color:' + t1 + '">'
      + Number(val).toLocaleString() + '</span></div>';
  });

  return '<div style="background:' + bg
    + ';border:1px solid ' + border
    + ';border-radius:10px;padding:10px 14px;'
    + 'min-width:180px;box-shadow:' + shadow + '">'
    + '<div style="font-size:11px;color:' + t4
    + ';font-family:DM Mono,monospace;margin-bottom:6px">'
    + label + '</div>' + rows + '</div>';
}
Chart do's & don'ts
Use the shared base config — never configure fonts or grid per-chart
Use gradient fill on area charts — opacityFrom: 0.2, opacityTo: 0
Use DM Mono for axis labels, tooltip values, and data labels
Use borderRadius: 5 on bar charts with borderRadiusApplication: 'end'
Use donut (not pie) — size: '72%' for the hole
Pass isDark flag and swap CHART_COLORS ↔ CHART_COLORS_DARK
Show the ApexCharts toolbar — always toolbar: { show: false }
Use 3D effects, drop shadows, or outer glow on chart elements
Use more than 5 colors in a single chart — group into "Other"
Use filled pie — always donut. Filled pie looks heavy
Show markers by default on line charts — only on hover

14Do's & don'ts
Quick reference for the most common mistakes.
Use warm off-white page background (#F8F8F6), not pure white or cold gray
Use font-mono for all numbers, IDs, timestamps, and tracking codes
Use 1px border at 6% opacity for card borders — no box-shadow
Use rounded-[14px] for cards, rounded-[10px] for inputs, rounded-lg for buttons
Use uppercase + tracking-wider + text-[10px] for table headers
Use dot + label for status badges, not colored background alone
Always test both light and dark — use the dark: prefix consistently
Use font-bold (700) anywhere — max is font-semibold (600)
Use heavy borders between table rows — use 1px at 6% opacity
Use box-shadow on cards — only on floating panels
Use pure black (#000) or white (#FFF) for dark mode backgrounds
Use spinners for loading — use skeleton placeholders instead
Mix font-sans and font-mono in the same label
Use more than 2 accent colors per view besides neutral
